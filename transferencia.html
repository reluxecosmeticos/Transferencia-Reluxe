  <script>

    const data = {

      nombre: "Reluxe Cosméticos Ltda.",

      rut: "772534817",

      email: "pagos@reluxe.cl",

      cuenta: "21670189110"

    };


    const toastEl = document.getElementById("toast");

    const allTextEl = document.getElementById("allText");


    function buildAllText(){

      return [

        `Nombre: ${data.nombre}`,

        `RUT: ${data.rut}`,

        `Email: ${data.email}`,

        `Número de cuenta: ${data.cuenta}`

      ].join("\n");

    }


    function showToast(msg){

      toastEl.textContent = msg;

      clearTimeout(showToast._t);

      showToast._t = setTimeout(() => toastEl.textContent = "", 2500);

    }


    async function copyText(text){

      // Prefer Clipboard API (best on iPhone Safari when triggered by a click)

      try{

        await navigator.clipboard.writeText(text);

        return true;

      }catch(e){

        // Fallback to execCommand

        try{

          const ta = document.createElement("textarea");

          ta.value = text;

          ta.style.position = "fixed";

          ta.style.left = "-9999px";

          ta.style.top = "-9999px";

          document.body.appendChild(ta);

          ta.focus();

          ta.select();

          const ok = document.execCommand("copy");

          document.body.removeChild(ta);

          return ok;

        }catch(_){

          return false;

        }

      }

    }


    async function copyOne(id){

      const text = data[id] ?? document.getElementById(id)?.textContent ?? "";

      const ok = await copyText(text);

      showToast(ok ? "✅ Copiado al portapapeles" : "⚠️ No se pudo copiar. Usa 'Seleccionar texto'.");

    }


    async function copyAll(){

      const text = buildAllText();

      const ok = await copyText(text);

      showToast(ok ? "✅ Datos completos copiados" : "⚠️ No se pudo copiar. Usa 'Seleccionar texto'.");

    }


    function selectAllText(){

      allTextEl.focus();

      allTextEl.select();

      showToast("Seleccionado. Ahora toca 'Copiar'.");

    }


    function openMail(){

      window.location.href = `mailto:${data.email}`;

    }


    // Init textarea

    allTextEl.value = buildAllText();
